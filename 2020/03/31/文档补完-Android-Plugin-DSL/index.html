文档补完-Android Plugin DSL

Android Plugin 3.4.0-dev DSL Reference

apply plugin: 'com.android.application'
apply plugin: 'kotlin-android'
apply plugin: 'kotlin-android-extensions'

android {
    compileSdkVersion 29
    buildToolsVersion "29.0.2"

    defaultConfig {
        applicationId "me.aoyun.customfancontroller"
        minSdkVersion 19
        targetSdkVersion 29
        versionCode 1
        versionName "1.0"

        testInstrumentationRunner "androidx.test.runner.AndroidJUnitRunner"
    }

    buildTypes {
        release {
            minifyEnabled false
            proguardFiles getDefaultProguardFile('proguard-android-optimize.txt'), 'proguard-rules.pro'
        }
    }

    compileOptions {
        sourceCompatibility JavaVersion.VERSION_1_8
        targetCompatibility JavaVersion.VERSION_1_8
    }

    kotlinOptions {
        jvmTarget = JavaVersion.VERSION_1_8.toString()
    }
}

dependencies {
    implementation fileTree(dir: 'libs', include: ['*.jar'])
    implementation "org.jetbrains.kotlin:kotlin-stdlib-jdk7:$kotlin_version"
    implementation 'androidx.appcompat:appcompat:1.1.0'
    implementation 'androidx.core:core-ktx:1.2.0'
    implementation 'androidx.constraintlayout:constraintlayout:1.1.3'
    testImplementation 'junit:junit:4.12'
    androidTestImplementation 'androidx.test.ext:junit:1.1.1'
    androidTestImplementation 'androidx.test.espresso:espresso-core:3.2.0'
}

defaultConfig{}

指定编译构建变体的默认变体。

testInstrumentationRunner Test instrumentation runner 类名。

buildType{}

配置构建类型。

minifyEnabled 是否移除未使用的 Java 代码。

proguardFiles 指定 plugin 应该使用的混淆文件。

compileOptions{}

sourceCompatibility Java 源代码的语言版本。

支持的格式：

  • "1.6"
  • 1.6
  • JavaVersion.Version_1_6
  • "Version_1_6"

targetCompatibility 生成的字节码的版本。

支持格式同上。
